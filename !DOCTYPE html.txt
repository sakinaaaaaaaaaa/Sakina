<script>
    function calculateExpenditure() {
        // Get input values from user
        const income = parseFloat(document.getElementById('income').value);
        const familySize = parseInt(document.getElementById('familySize').value);
        
        // Validate inputs
        if (isNaN(income) || isNaN(familySize)) {
            alert("Please enter valid numbers for both fields.");
            return;
        }
        
        // Regression model formula calculation
        // Intercept = -1.1183, Income Coeff = 0.1482, Family Size Coeff = 0.7931
        let predictedFood = -1.1183 + (0.1482 * income) + (0.7931 * familySize);
        
        // Handle lower bound edge-cases dynamically (expenditure cannot be negative)
        if (predictedFood < 0) {
            predictedFood = 0;
        }
        
        // Display result container
        const resultDiv = document.getElementById('result');
        resultDiv.style.display = 'block';
        
        // Format result to 4 decimal places
        resultDiv.innerHTML = `Predicted Food Expenditure:<br><span style="color: #28a745; font-size: 24px;">RM ${predictedFood.toFixed(4)}</span>`;
    }
</script>
